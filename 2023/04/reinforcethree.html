def e_greedy_policy(state,epsilon=0): # epsilon-greedy policy
    if np.random.rand() < epsilon: # exploration
        return np.random.randint(3) # actionì„ ëœë¤í•˜ê²Œ ì„ íƒ
    else: # exploitation
        input_load = state[0:inputlen_load].reshape(-1,inputlen_load,1) # Conv1Dì˜ inputì´ë¯€ë¡œ ì±„ë„ ìˆ˜ 1 ëª…ì‹œ
        input_pv = state[inputlen_load:(inputlen_load+inputlen_pv)].reshape(-1,inputlen_pv,1)
        input_others =state[(inputlen_load+inputlen_pv)].reshape(-1,1) # Denseì¸µì˜ inputì´ë¯€ë¡œ ì±„ë„ ìˆ˜ëŠ” í•„ìš” ì—†ìŒ
        Q_values = model((input_load,input_pv,input_others)) # ê° action ë³„ Q-value ë„ì¶œ
        return np.argmax(Q_values[0]) # ê°€ì¥ í° Q-valueì— ëŒ€ì‘í•˜ëŠ” action ì„ íƒ      
        # ì£¼ì˜: for loop ì•ˆì—ì„œ DNNì— inputì„ ì…ë ¥í•´ output ê³„ì‚° ì‹œ model()ë¡œ í•´ì•¼ì§€, model.predict()ë¡œ í•˜ë©´ ì•ˆ ë¨! ë©”ëª¨ë¦¬ ëˆ„ìˆ˜ê°€ ë°œìƒí•¨ (model.predictëŠ” ëŒ€ëŸ‰ì˜ input dataë¥¼ 'model.predictë¥¼ í•œ ë²ˆë§Œ í˜¸ì¶œí•´ì„œ' ì²˜ë¦¬í•˜ëŠ” ë° íŠ¹í™”ë¨, https://www.tensorflow.org/api_docs/python/tf/keras/Model#predict ì°¸ê³ )



from collections import deque
replay_buffer = deque(maxlen=buffer_size) # replay buffer ì •ì˜ì˜

def sample_experiences(batch_size): # batch_sizeë§Œí¼ì˜ ê²½í—˜ë“¤ì˜ stateë“¤, actionë“¤, rewardë“¤, nextstateë“¤ì˜ ë¦¬ìŠ¤íŠ¸ë“¤ì„ ë°˜í™˜
    indices = np.random.randint(len(replay_buffer), size=batch_size) # replay buffer ë‚´ ê²½í—˜ë“¤ ì¤‘ ëœë¤í•˜ê²Œ batch_sizeë§Œí¼ì˜ ê²½í—˜ë“¤ì„ ì§€ì • (ì¸ë±ìŠ¤ ë¶ˆëŸ¬ì˜´)
    batch = [replay_buffer[index] for index in indices] # ìœ„ì—ì„œ ë¶ˆëŸ¬ì˜¨ ì¸ë±ìŠ¤ë¥¼ ì´ìš©í•´ batch_size ë§Œí¼ì˜ ê²½í—˜ë“¤ì„ batch ë¦¬ìŠ¤íŠ¸ì— ë‹´ìŒ
    states, actions, rewards, next_states = [
        np.array([experience[field_index] for experience in batch]) # í•˜ë‚˜ì˜ array ì•ˆì— batch ë‚´ ê° ê²½í—˜ë³„ ê°’ë“¤ì´ ë“¤ì–´ê° 
        for field_index in range(4)] # ì´ 4ê°œì˜ arrayë¥¼ ë°˜í™˜, ê°ê°ì€ (batch ë‚´ sampleê²½í—˜ë“¤ì˜) stateë“¤, actionë“¤, rewardë“¤, nextstateë“¤ì˜ ë¦¬ìŠ¤íŠ¸ì„
    return states, actions, rewards, next_states



def play_one_step(profile_load,profile_pv,hour,energy_batt,epsilon,training=False):
        
    state = np.concatenate( (profile_load[hour-inputlen_load:hour], profile_pv[hour-inputlen_pv:hour], np.array([energy_batt])) )
    
    action = e_greedy_policy(state,epsilon) # epsilon-greedy policyì— ë”°ë¼ action ë„ì¶œ
    p_h2_send = 0
    p_h2_receive = 0
    if action == 0: # ìˆ˜ì†Œë¡œ ìƒì‚°ëœ ì „ê¸° ìˆ˜ì „
        p_h2_receive = prate_h2
    elif action == 1: # ìˆ˜ì†Œ ìƒì‚°ìš© ì „ê¸° ì†¡ì „
        p_h2_send = prate_h2
       

    p_load = profile_load[hour]*load_peak # í˜„ì¬ ì‹œê°„ì˜ ë¶€í•˜ (action ê²°ì • ê¸°ì¤€ì´ ì•„ë‹˜!), ì…ë ¥ìë£Œê°€ [0,1]ë¡œ scaledëœ ê±¸ unscaling
    p_pv = profile_pv[hour]*pv_peak # í˜„ì¬ ì‹œê°„ì˜ ë°œì „ëŸ‰ (action ê²°ì • ê¸°ì¤€ì´ ì•„ë‹˜!), ì…ë ¥ìë£Œê°€ [0,1]ë¡œ scaledëœ ê±¸ unscaling  
    energy_batt = energy_batt*capa_batt # ì…ë ¥ìë£Œê°€ [0,1]ë¡œ scaledëœ ê±¸ unscaling
    
    #p_curtail = 0
    p_loss = 0
    
    p_net_beforebatt = p_pv - p_load + p_h2_receive - p_h2_send # íƒœì–‘ê´‘ ìƒì‚°, ë¶€í•˜ ì¶©ì¡±, H2 ë³´ë‚´ê±°ë‚˜ ë°›ì€ í›„ ìˆ˜ìš©ê°€ ì…ì¥ì—ì„œ ì „ê¸°ì—ë„ˆì§€ê°€ ë‚¨ìœ¼ë©´ ì–‘ìˆ˜, ë¶€ì¡±í•˜ë©´ ìŒìˆ˜
    
    if p_net_beforebatt >= 0: # ì „ê¸°ì—ë„ˆì§€ê°€ ë‚¨ìœ¼ë¯€ë¡œ ë°°í„°ë¦¬ì— ì €ì¥í•˜ê³ , ë§Œì•½ ë°°í„°ë¦¬ë„ ê½‰ ì°¬ë‹¤ë©´ curtailí•¨
        if capa_batt >= energy_batt + p_net_beforebatt*eff_batt: # ë°°í„°ë¦¬ì— ì”ì—¬ ì¶©ì „ ê°€ëŠ¥í•œ ì—ë„ˆì§€ê°€ ìœ„ì—ì„œ ë‚¨ì€ ì—ë„ˆì§€(ì—ì„œ ë³€í™˜ì†ì‹¤ ì œí•œ ì—ë„ˆì§€) ì´ìƒì¼ ê²½ìš°
            energy_batt_after = energy_batt + p_net_beforebatt*eff_batt
        else: # ë‚¨ì€ ì—ë„ˆì§€ë¥¼ ë°°í„°ë¦¬ì— ë‹¤ ì¶©ì „í•˜ë©´ ê½‰ ì°¨ê³ ë„ ë‚¨ì•„ì„œ, curtailí•´ì•¼ í•¨
            energy_batt_after = capa_batt
            #p_curtail = (energy_batt + p_net_beforebatt*eff_batt - capa_batt)/eff_batt # ê´„í˜¸ ì•ˆì€ ë°°í„°ë¦¬ ë‚´ë¶€ ê¸°ì¤€ì´ë©°, ìˆ˜ìš©ê°€ ëª¨ì„  ê¸°ì¤€ ì–‘ì„ êµ¬í•˜ë ¤ë©´ eff_battë¡œ ë‚˜ëˆ ì¤˜ì•¼ í•¨
    else: # ì „ê¸°ì—ë„ˆì§€ê°€ ë¶€ì¡±í•˜ë¯€ë¡œ ë°°í„°ë¦¬ ì—ë„ˆì§€ë¥¼ ì¨ì•¼ í•¨, ë°°í„°ë¦¬ ì—ë„ˆì§€ë¡œë„ ë¶€ì¡±í•˜ë©´ lossì„
        if energy_batt*eff_batt >= -p_net_beforebatt: # ë°°í„°ë¦¬ ì—ë„ˆì§€ë¡œ ì¶©ë‹¹ ê°€ëŠ¥í•œ ê²½ìš°
            energy_batt_after = energy_batt + p_net_beforebatt/eff_batt # p_net_beforebattê°€ ìŒìˆ˜ì´ë¯€ë¡œ ë§ˆì´ë„ˆìŠ¤ê°’ì´ ë¶€ì¡±í•œ ì—ë„ˆì§€'ëŸ‰'ì´ê³  ê·¸ê±¸ 'ë¹¼'ë¯€ë¡œ ê²°ê³¼ì ìœ¼ë¡œ í”ŒëŸ¬ìŠ¤
        else: # ë¶€ì¡±í•´ loss ë°œìƒ
            energy_batt_after = 0
            p_loss = -p_net_beforebatt - energy_batt*eff_batt           
    
    reward = price_h2*p_h2_send*eff_h2 - price_h2*p_h2_receive/eff_h2 - cost_loss*p_loss
    energy_batt_after = energy_batt_after/capa_batt # ë°°í„°ë¦¬ ë‚´ ì €ì¥ëŸ‰ì„ [0,1] ë²”ìœ„ë¡œ scalingí•¨
    
    if training == True:
        next_state = np.concatenate( (profile_load[hour-(inputlen_load-1):hour+1], profile_pv[hour-(inputlen_pv-1):hour+1], np.array([energy_batt_after])) )         
        replay_buffer.append((state, action, reward*rewardscalefactor, next_state)) # replay bufferì—ëŠ” scaled rewardë¥¼ ë„£ìœ¼ë©°, tupleì„ appendí•¨ì— ì£¼ì˜
    return energy_batt_after, reward, action # scaled ë°°í„°ë¦¬ ë‚´ ì €ì¥ëŸ‰, unscaled reward, action index ë°˜í™˜



def training_step(batch_size):

    states, actions, rewards, next_states = sample_experiences(batch_size)
        
    input_load = states[:,0:inputlen_load].reshape(-1,inputlen_load,1) 
    input_pv = states[:,inputlen_load:(inputlen_load+inputlen_pv)].reshape(-1,inputlen_pv,1)
    input_others = states[:,(inputlen_load+inputlen_pv)].reshape(-1,1)
    
    input_load_next = next_states[:,0:inputlen_load].reshape(-1,inputlen_load,1)
    input_pv_next = next_states[:,inputlen_load:(inputlen_load+inputlen_pv)].reshape(-1,inputlen_pv,1)
    input_others_next = next_states[:,(inputlen_load+inputlen_pv)].reshape(-1,1)
      
    next_Q_values = model((input_load_next,input_pv_next,input_others_next)) # Q-learningì„ ìœ„í•´, 'next'stateì—ì„œì˜ ê° action ë³„ Q-value ì¶”ì •ì¹˜ë“¤ ë°˜í™˜
    max_next_Q_values = np.max(next_Q_values, axis=1) # ê° ê²½í—˜ë³„ë¡œ nextstateì— ëŒ€í•œ Q-valueê°€ ë” ë†’ì€ í–‰ë™ì˜ Q-value ì‚¬ìš©
    target_Q_values = (rewards + discount_factor * max_next_Q_values) # Q-valueë¥¼ rewardì™€ nextstateì— ëŒ€í•œ Q-value(discounted)ì˜ í•©ìœ¼ë¡œ í‘œí˜„
    target_Q_values = target_Q_values.reshape(-1,1) # ë’¤ì˜ Q_valuesì™€ ì°¨ì› ë§ì¶°ì¤Œ
    mask = tf.one_hot(actions, n_outputs) # ê° ê²½í—˜ ë³„ actionì„ one-hot encoding í˜•íƒœë¡œ ë§Œë“¤ì–´ì¤Œ (ê° í–‰ì´ ê²½í—˜)
    with tf.GradientTape() as tape: # ìë™ ë¯¸ë¶„
        all_Q_values = model((input_load,input_pv,input_others)) 
        Q_values = tf.reduce_sum(all_Q_values * mask, # maskë¥¼ ê³±í•´ì„œ, ê° ê²½í—˜ë³„ë¡œ ê·¸ stateì—ì„œ ì·¨í•˜ì§€ ì•Šì€ actionì— ëŒ€í•´ì„œëŠ” Q-valueì— 0ì´ ê³±í•´ì§€ë„ë¡ í•¨
                                 axis=1, keepdims=True) # ê° í–‰ ë³„ í•©ì„ êµ¬í•¨, ìœ„ì˜ masking ë•ë¶„ì— ê·¸ stateì—ì„œ ì·¨í•œ actionì— ëŒ€í•œ Q-valueê°€ ë¨, keepdimsë¥¼ Trueë¡œ ì„¤ì •í•´ 2ì°¨ì› í–‰ë ¬ í˜•íƒœ ìœ ì§€
        loss = tf.reduce_mean(keras.losses.mean_squared_error(target_Q_values,Q_values)) # í‰ê· ì œê³±ì˜¤ì°¨ ê³„ì‚°
    grads = tape.gradient(loss, model.trainable_variables) # loss í•¨ìˆ˜ë¥¼ modelì˜ trainable variables ì „ì²´ì— ëŒ€í•´ ë¯¸ë¶„
    optimizer.apply_gradients(zip(grads, model.trainable_variables)) # Adam optimizerë¡œ parameter update ìˆ˜í–‰, zipìœ¼ë¡œ gradientì™€ parameter pairë¥¼ ë§ì¶°ì¤Œ



profits_test = []
elapsedtime_test = []
count_step_fortrain = 0
