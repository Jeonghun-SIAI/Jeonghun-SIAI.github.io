def play_one_step(profile_load,profile_pv,hour,energy_batt,epsilon,training=False):
        
    state = np.concatenate( (profile_load[hour-inputlen_load:hour], profile_pv[hour-inputlen_pv:hour], np.array([energy_batt])) )
     
    # profile_load = load_train #(ì½”ë“œ ì˜ë„ëŠ”ì§€ í…ŒìŠ¤íŠ¸ìš©)

    if np.random.rand() < epsilon and training == True: # exploration during training
        action = np.random.uniform(low=-1.0, high=1.0, size=1)[0] # exploration ì‹œ -1~+1 ì‚¬ì´ì˜ ì‹¤ìˆ˜ë¥¼ ê· ë“±ë¶„í¬ì—ì„œ ì¶”ì¶œ

    p_load = profile_load[hour]*load_peak # í˜„ì¬ ì‹œê°„ì˜ ë¶€í•˜ (action ê²°ì • ê¸°ì¤€ì´ ì•„ë‹˜!), ì…ë ¥ìë£Œê°€ [0,1]ë¡œ scaledëœ ê±¸ unscalingí•¨

    energy_batt = energy_batt*capa_batt # unscaling
    
    
    #p_curtail = 0
    p_loss = 0
    
    p_net_beforebatt = p_pv - p_load + p_h2_receive - p_h2_send # íƒœì–‘ê´‘ ìƒì‚°, ë¶€í•˜ ì¶©ì¡±, H2 ë³´ë‚´ê±°ë‚˜ ë°›ì€ í›„ ìˆ˜ìš©ê°€ ì…ì¥ì—ì„œ ì „ê¸°ì—ë„ˆì§€ê°€ ë‚¨ìœ¼ë©´ ì–‘ìˆ˜, ë¶€ì¡±í•˜ë©´ ìŒìˆ˜

        if capa_batt >= energy_batt + p_net_beforebatt*eff_batt: # ë°°í„°ë¦¬ì— ì”ì—¬ ì¶©ì „ ê°€ëŠ¥í•œ ì—ë„ˆì§€ê°€ ìœ„ì—ì„œ ë‚¨ì€ ì—ë„ˆì§€(ì—ì„œ ë³€í™˜ì†ì‹¤ ì œí•œ ì—ë„ˆì§€) ì´ìƒì¼ ê²½ìš°
            energy_batt_after = energy_batt + p_net_beforebatt*eff_batt
        else: # ë‚¨ì€ ì—ë„ˆì§€ë¥¼ ë°°í„°ë¦¬ì— ë‹¤ ì¶©ì „í•˜ë©´ ê½‰ ì°¨ê³ ë„ ë‚¨ì•„ì„œ, curtailí•´ì•¼ í•¨
            energy_batt_after = capa_batt
            # p_curtail = (energy_batt + p_net_beforebatt*eff_batt - capa_batt)/eff_batt # ê´„í˜¸ ì•ˆì€ ë°°í„°ë¦¬ ë‚´ë¶€ ê¸°ì¤€ì´ë©°, ìˆ˜ìš©ê°€ ëª¨ì„  ê¸°ì¤€ ì–‘ì„ êµ¬í•˜ë ¤ë©´ eff_battë¡œ ë‚˜ëˆ ì¤˜ì•¼ í•¨
    else: # ì „ê¸°ì—ë„ˆì§€ê°€ ë¶€ì¡±í•˜ë¯€ë¡œ ë°°í„°ë¦¬ ì—ë„ˆì§€ë¥¼ ì¨ì•¼ í•¨, ë°°í„°ë¦¬ ì—ë„ˆì§€ë¡œë„ ë¶€ì¡±í•˜ë©´ lossì„
        if energy_batt*eff_batt >= -p_net_beforebatt: # ë°°í„°ë¦¬ ì—ë„ˆì§€ë¡œ ì¶©ë‹¹ ê°€ëŠ¥í•œ ê²½ìš°
            energy_batt_after = energy_batt + p_net_beforebatt/eff_batt # p_net_beforebattê°€ ìŒìˆ˜ì´ë¯€ë¡œ ë§ˆì´ë„ˆìŠ¤ê°’ì´ ë¶€ì¡±í•œ ì—ë„ˆì§€'ëŸ‰'ì´ê³  ê·¸ê±¸ 'ë¹¼'ë¯€ë¡œ ê²°ê³¼ì ìœ¼ë¡œ í”ŒëŸ¬ìŠ¤
        else: # ë¶€ì¡±í•´ loss ë°œìƒ
            energy_batt_after = 0
            p_loss = -p_net_beforebatt - energy_batt*eff_batt           
    
    reward = price_h2*p_h2_send*eff_h2 - price_h2*p_h2_receive/eff_h2 - cost_loss*p_loss # rewardë„ scalingí•¨

        replay_buffer.append((state, action, reward*rewardscalefactor, next_state)) # tupleì„ appendí•¨ì— ì£¼ì˜
    return energy_batt_after, reward, action



def training_step(batch_size,actorupdate=True):
    
    states, actions, rewards, next_states = sample_experiences(batch_size)
        
    input_load = states[:,0:inputlen_load].reshape(-1,inputlen_load,1) 
    input_pv = states[:,inputlen_load:(inputlen_load+inputlen_pv)].reshape(-1,inputlen_pv,1)

class Actor(tf.keras.Model): # (Continuous) Actionì„ ê²°ì •í•˜ëŠ” policy NN
    def __init__(self):
        super(Actor, self).__init__() # tf.Keras.Model ìƒì†ë°›ëŠ” ì´ìœ ëŠ” mainì½”ë“œì—ì„œ get_weights, set_weights ë“±ì„ ë§¤ë²ˆ model ë©”ì†Œë“œë¥¼ ë¶ˆëŸ¬ì˜¤ì§€ ì•Šê³ ë„ í¸í•˜ê²Œ ì“°ê¸° ìœ„í•¨          
        self.input_load = Input(shape=(inputlen_load,1)) # ê³¼ê±° 12ì‹œê°„ì˜ ë¶€í•˜, shapeì˜ ë‘ë²ˆì§¸ ìˆ«ìëŠ” ì±„ë„ ìˆ˜ (ì»¬ëŸ¬ì‚¬ì§„ì˜ RGB ê°™ì€), ì´ê²Œ ìˆì–´ì•¼ Conv1Dê°€ ì‘ë™í•¨
        self.input_pv = Input(shape=(inputlen_pv,1)) # ê³¼ê±° 12ì‹œê°„ì˜ íƒœì–‘ê´‘ë°œì „ëŸ‰
        self.hidden_conv_load = Conv1D(16, 2, activation='relu')(self.input_load)
        self.hidden_conv_pv = Conv1D(16, 2, activation='relu')(self.input_pv)
        self.concat_conv = concatenate([self.hidden_conv_load,self.hidden_conv_pv],axis=2)
        self.hidden_conv_concat = Conv1D(16,2,activation='relu')(self.concat_conv)
        self.flatten_conv = Flatten()(self.hidden_conv_concat) # Denseì¸µ ì§ì „ì—ëŠ” Convì¸µì„ Flattení•˜ëŠ” ê²ƒ ëª…ì‹¬
        self.input_others = Input(shape=(1)) # í˜„ì¬ ë°°í„°ë¦¬ ë‚´ ì—ë„ˆì§€ëŸ‰
        self.concat_all = concatenate([self.flatten_conv,self.input_others]) 
        self.hidden_dense_1 = Dense(50,activation='relu')(self.concat_all)
        self.hidden_dense_2 = Dense(20,activation='relu')(self.hidden_dense_1)
        self.output_action_mean = Dense(1,activation='linear')(self.hidden_dense_2) # ì •ê·œë¶„í¬ì˜ í‰ê· 
        self.output_action_stddev = Dense(1,activation='linear')(self.hidden_dense_2) # ì •ê·œë¶„í¬ì˜ í‘œì¤€í¸ì°¨
        self.model = keras.Model(inputs=[self.input_load,self.input_pv,self.input_others], outputs=[self.output_action_mean,self.output_action_stddev])     
    def call(self, input_load,input_pv,input_others,training=True): # stateê°€ inputì„        
        mu, logsigma = self.model((input_load,input_pv,input_others))
        sigma = tf.exp(logsigma)
        dist = tfp.distributions.Normal(mu, sigma)
        if training == True:
            action_temp = dist.sample()
        else:
            action_temp = mu # test ì‹œì—ëŠ” mean actionìœ¼ë¡œ (deterministic actionìœ¼ë¡œ testí•˜ë¯€ë¡œ)
        action = tf.tanh(action_temp) # bounded action êµ¬í˜„
        logprob_temp = dist.log_prob(action_temp)
        logprob = logprob_temp - tf.reduce_sum(tf.math.log(1 - action**2 + 1e-16), axis=1,
                                         keepdims=True) # log-probability ë°˜í™˜
        return action, logprob
